import numpy as np
import seaborn as sb
import pandas as pd

WORLD_SIZE = 10

def sample_reward_scheme():
    return { 
        "bird_reward" : (np.random.rand() - 0.5) * 40 - 10,
        "flower_reward" : (np.random.rand() - 0.5) * 2 
    }

def sample_landscape():
    return {
        "bird_probabilities" : np.random.rand(WORLD_SIZE),
        "flower_quantities": np.random.rand(WORLD_SIZE) * 10
    }

## returns (avg fitness, avg reward) 
def evaluate_reward_scheme_on_landscapes(reward_scheme, landscapes):
    total_fitness, total_reward = (0, 0)
    for landscape in landscapes:
        this_fitness, this_reward = evaluate_reward_scheme_on_landscape(reward_scheme, landscape)
        total_fitness += this_fitness
        total_reward += this_reward
    return total_fitness/len(landscapes), total_reward/len(landscapes)
        
default_model_params = {
    "lifetime" : 100,
    "epsilon": 0.1,
    "bird_eating_prob": 0.05, # reasonable is 0.05
    "contact_with_bird_from_getting_eaten": 0,
    "metabolic_cost_coefficient": 0.05
}

def repeatedly_evaluate_reward_scheme_on_landscape(reward_scheme, landscape, model_params=default_model_params):
    runs = pd.DataFrame([evaluate_reward_scheme_on_landscape(reward_scheme, landscape) for _ in range(50)])
    return {
        "fitness": runs["fitness"].mean(),
        "total_reward": runs["total_reward"].mean(),
        "total_flowers": runs["total_flowers"].mean(),
        "flower_reward": reward_scheme["flower_reward"],
        "bird_reward": reward_scheme["bird_reward"],
        "total_metabolic_reward_cost": runs["total_metabolic_reward_cost"].mean(),
        "eaten_by_bird": runs["eaten_by_bird"].mean(),
        
    }

# returns (fitness, total reward)
def evaluate_reward_scheme_on_landscape(reward_scheme, landscape, model_params=default_model_params):
    # Start out with an optimistic bias, to encourage exploration. TODO: is that dumb?
    beliefs = np.zeros(WORLD_SIZE) + 10
    counts = np.zeros(WORLD_SIZE) + 1
    total_reward = 0
    total_flowers = 0
    total_metabolic_reward_cost = 0
    
    world_size = landscape["bird_probabilities"].size
    
    bird_reward = reward_scheme["bird_reward"]
    flower_reward = reward_scheme["flower_reward"]
    
    def return_dict():
        return  { 
            "fitness": fitness, 
            "total_reward": total_reward,
            "total_flowers": total_flowers,
            "flower_reward": flower_reward,
            "bird_reward": bird_reward,
            "total_metabolic_reward_cost": total_metabolic_reward_cost,
            "eaten_by_bird": eaten_by_bird
        }
    
    
    for step in range(model_params["lifetime"]):
        # choose an action:
        if np.random.rand() < model_params["epsilon"]:
            # act randomly
            action = np.random.randint(world_size)
        else:
            # do best action
            action = np.argmax(beliefs)
        
        bird_prob = landscape["bird_probabilities"][action]
        flower_quantity = landscape["flower_quantities"][action]
        
        # We're going to accumulate the reward from this action here. 
        # We initialize it with some Gaussian noise with std dev 0.1.
        reward_from_this_action = np.random.normal(scale=0.1)
        
        # do we get eaten by a bird?
        if np.random.rand() < bird_prob * model_params["bird_eating_prob"]:
            # oh no :(
            total_reward += bird_reward * model_params["contact_with_bird_from_getting_eaten"]
                
            # You got eaten, so fitness is zero.
            fitness = 0
            eaten_by_bird = True
            return return_dict()

        # How much were we scared by the evidence of nearby bird activity?
        this_bird_reward = bird_reward * bird_prob
        reward_from_this_action += this_bird_reward 
        total_metabolic_reward_cost += this_bird_reward ** 2
        
        # now we eat flowers!
        flower_reward_from_this_action = flower_quantity * flower_reward
        total_metabolic_reward_cost += flower_reward_from_this_action ** 2
        reward_from_this_action += flower_reward_from_this_action
        
        total_flowers += flower_quantity
        
        # now we update our beliefs!
        # TODO: I should probably do this by taking a real average rather than a running average
        counts[action] += 1
        alpha = 1 / counts[action]
        beliefs[action] = alpha * reward_from_this_action + (1 - alpha) * beliefs[action]
        total_reward += reward_from_this_action
        
    # phew, we made it all the way to adulthood, so we can made and have babies proportional 
    # to how many flowers we got minus how much energy we spent on being happy or sad!
    fitness = max(total_flowers - total_metabolic_reward_cost * model_params["metabolic_cost_coefficient"], 0)
    eaten_by_bird = False
    return return_dict()
